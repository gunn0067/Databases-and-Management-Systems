4. design a trigger which will be fired when a new record is inserted or updated for the following conditions:
   1. salary shold be more than 10000 for category 1
   2. salary should be more than 20000 for category 2
   3. salary should be more than 30000 for category 3

CREATE OR REPLACE TRIGGER emp2_salary_check
BEFORE INSERT OR UPDATE ON emp2
FOR EACH ROW
BEGIN
    IF :NEW.category = 1 AND :NEW.salary <= 10000 THEN
        RAISE_APPLICATION_ERROR(-20001, 'Salary must be greater than 10000 for category 1');

    ELSIF :NEW.category = 2 AND :NEW.salary <= 20000 THEN
        RAISE_APPLICATION_ERROR(-20002, 'Salary must be greater than 20000 for category 2');

    ELSIF :NEW.category = 3 AND :NEW.salary <= 30000 THEN
        RAISE_APPLICATION_ERROR(-20003, 'Salary must be greater than 30000 for category 3');
    END IF;
END;
/
