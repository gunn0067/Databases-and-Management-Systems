    deptno INT PRIMARY KEY,
    dept_name VARCHAR(50),

Employee Table


CREATE TABLE emp(
    emp_no INT PRIMARY KEY,
    birth_date DATE NOT NULL,
    first_name VARCHAR(14) NOT NULL,
    last_name VARCHAR(16) NOT NULL,
    gender varchar2(1) check (gender in ('M','F')) NOT NULL,
    hire_date DATE NOT NULL,
    salary FLOAT DEFAULT 7850.00,
    job VARCHAR(20) NOT NULL,
    deptno INT,
    manager_id INT,
    FOREIGN KEY (deptno) REFERENCES departments(deptno)
);

1. Display all employee names that starts with "a" and are at least 3 Characters in length. 
select first_name, last_name from emp where lower(first_name) like 'a%' and length (first_name) >4;

2. Display departments having Dallas, New York and Chicago locations. 
select * from departments where lower (location)='dallas' or lower(location) = 'new york' or lower(location)= 'chicago';

3. Display the names of employees who are working as Clerk; salesman or analyst and drawing a salary more than 3000. 
select first_name, job, salary from emp where lower(job) IN ('clerk', 'salesman', 'analyst') and salary>=3000;

4. display the names of employees who are not working as managers. 
select First_name, job from emp where lower(job)!='manager1';

5. Display the name of emp who earns highest salary

6. Display the employee number and name of employee working as CLERK and earning highest salary among CLERKS. 
select emp_no, first_name, salary from emp where salary= (select max(salary) from emp where lower(job) = 'clerk');

7. Display the names of clerks who earn salary lesser than that of James of that of salary more than that of Bob. 
select emp_no, first_name, salary from emp where lower(job)='clerk' and salary > (select salary from emp where lower(First_name)='Bob') and salary < (select salary from emp where lower(First_name)='james');

8. Display the name of the employees who earns highest salary in their respective departments. 
select i.first_name, i.last_name, i.salary from emp i where i.salary = (select max(i2.salary) from emp i2 where i2.deptno = i.deptno);
